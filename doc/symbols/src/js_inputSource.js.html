 53     /**
 54      * Import the video tag to show the input source dynamically. This is used to embed input source in a html page.
 55      *  
 56      * @class InputSource
 57      * @param {Function} successCallback success callback function.
 58      * @param {Function} errorCallback failure callback function.
 59      * @param {Object} options
 60      * <div align=left>
 61      * <table class="hcap_spec" width=400>
 62      *   <thead><tr><th>Property</th><th>Type</th><th>Description</th><th>Required</th></tr></thead>
 63      *   <tbody>
 64      *       <tr><th>divId</th><th>String</th><th>Div tag ID with the video tag in it. </th><th>required</th></tr>
 65      *       <tr class="odd"><th>videoId</th><th>String</th><th>The ID of the video tag to be dynamically created. </th><th>required</th></tr>
 66      *       <tr><th>callback</th><th>Function</th><th>An event handler (onloadedmetadata) that occurs while the video tag is being created. </th><th>required</th></tr>
 67      *       <tr class="odd"><th>src</th><th>String</th><th>Define the src attribute of video tag. ext://[externalInput][portNum]. (eg: "ext://hdmi:1") </th><th>required</th></tr>
 68      *   </tbody>
 69      * </table>
 70      * @return None.
 71      * @example
 72      * // HTML code
 73      *  < body onload="onLoad()" >
 74      *     < div id = "videoDiv" style="width:640px; height:380px" >
 75      *     < /div >
 76      *  < /body >
 77      *
 78      * // Javascript code
 79      * function init () {
 80      * }
 81      * function onLoad() {
 82      *     var options = {};
 83      *     options.divId = "videoDiv"; // It should be matched to div tag name in an html page.
 84      *     options.videoId = "broadcastvideo";
 85      *     options.callback = init;
 86      *     options.src = "ext://hdmi:1";
 87      *     
 88      *     function successCb(cbObject) {
 89      *         // Do something
 90      *     }
 91      *
 92      *     function failureCb(cbObject) {
 93      *         var errorCode = cbObject.errorCode;
 94      *         var errorText = cbObject.errorText;
 95      *         
 96      *         console.log ("Error Code [" + errorCode + "]: " + errorText);
 97      *     }
 98      *     var inputSource = new InputSource();
 99      *     inputSource.initialize(successCb, failureCb, options);
100      * }
101      * @since 1.0
102      */
103     InputSource.prototype.initialize = function (successCallback, errorCallback, options) {
104     
105         log("initialize: " + JSON.stringify(options));
106         
107         // bound check
108         if (options.divId === undefined || typeof options.divId !== 'string' || options.divId === null || options.divId.length <= 0 ||
109                 options.videoId === undefined || typeof options.videoId !== 'string' || options.videoId === null || options.videoId.length <= 0 ||
110                 options.callback === undefined || typeof options.callback !== 'function' ||
111                 options.src === undefined || typeof options.src !== 'string' || options.src === null || options.src.length <= 0 ) {
112             
113             if (typeof errorCallback === 'function') {
114                 var result = {};
115                 checkErrorCodeNText(result, "II", "InputSource.initialize returns failure. invalid parameters.");
116                 errorCallback(result);
117             }
118             
119             return;
120         }
121         
122         convertInputSource(options.src, function success(inputSource) {
123             // success callback
124             options.src = inputSource;
125             broadcast = new Broadcast(); // jshint ignore:line
126             
127             log("initialize: converted " + JSON.stringify(options));
128             
129             broadcast.initialize(successCallback, errorCallback, options);
130             
131         }, function failure(cbObj) {
132             // failure callback
133             log("initialize: failure " + JSON.stringify(cbObj));
134             
135             if (typeof errorCallback === 'function') {
136                 var result = {};
137                 checkErrorCodeNText(result, "II", "InputSource.initialize returns failure. invalid parameters.");
138                 errorCallback(result);
139             }
140         });
141         
142         log("initialize: Done");
143         
144     };

146     /**
147      * Changes the input source that viewer is currently watching. This method is used to change to other input mode after initializing the inputSource objcet (creating video tag. etc.) using initialize method.
148      * Available input types and number of input source for each input types may differ according to the model. <br>
149      * For example, LS55A model has the following inputs : (HDMI1, HDMI2, DP, DVI)
150      * @class InputSource
151      * @param {Function} successCallback success callback function.
152      * @param {Function} errorCallback failure callback function.
153      * @param {Object} options
154      * <div align=left>
155      * <table class="hcap_spec" width=400>
156      *   <thead><tr><th>Property</th><th>Type</th><th>Description</th><th>Required</th></tr></thead>
157      *   <tbody>
158      *       <tr><th>src</th><th>String</th><th>Define the src attribute of video tag. "ext://[externalinput]:[portNum]" (eg, "ext://dp:1")</a> </th><th>required</th></tr>
159      *   </tbody>
160      * </table>
161      * </div>
162      * @return If the method is successfully executed, call the success callback function without a parameter. If an error occurs while performing the method, the failure callback function is called and Failure Callback Object is passed as a parameter.
163      * @example
164      * // Javascript code
165      * function changeInputSource () {
166      *     
167      *     var options = {};
168      *     options.src = "ext://hdmi:1";
169      *
170      *     function successCb(cbObject) {
171      *         // Do something
172      *     }
173      *
174      *     function failureCb(cbObject) {
175      *         var errorCode = cbObject.errorCode;
176      *         var errorText = cbObject.errorText;
177      *         
178      *         console.log ("Error Code [" + errorCode + "]: " + errorText);
179      *     }
180      *
181      *     var inputSource = new InputSource();
182      *     inputSource.changeInputSource(successCb, failureCb, options);
183      * }
184      * @since 1.0
185      * @see
186      */
187     InputSource.prototype.changeInputSource = function (successCallback, errorCallback, options) {
188     
189         // bound check
190         if ( options.src === undefined || typeof options.src !== 'string' || options.src === null || options.src.length <= 0) {
191             
192             if (typeof errorCallback === 'function') {
193                 var result = {};
194                 checkErrorCodeNText(result, "ICIS", "InputSource.changeInputSource returns failure. invalid argument. ");
195                 errorCallback(result);
196             }
197             
198             return;
199         }
200         
201        //TODO; deprecated changeInput in broadcast.js 
202         
203         log("changeInputSource: " + JSON.stringify(options));
204         
205         convertInputSource(options.src, function success (inputSource) {
206             // success callback
207             options.src = inputSource;
208             
209             if (broadcast.setInput(options)) {
210                 log("changeInputSource: On Success");
211                 successCallback();
212             } else {
213                 var result = {};
214                 log("changeInputSource: On Failure");
215                 checkErrorCodeNText(result, "ICIS", "InputSource.changeInputSource returns failure.");
216                 errorCallback(result);
217             }
218             
219         }, function failure(cbObj) {
220             // failure callback
221             log("changeInputSource: failure " + JSON.stringify(cbObj));
222             
223             if (typeof errorCallback === 'function') {
224                 var result = {};
225                 checkErrorCodeNText(result, "ICIS", "InputSource.changeInputSource returns failure. invalid argument. ");
226                 errorCallback(result);
227             }
228         });
229         
230         log("changeInputSource: Done");
231     };

238      * @return {Object} array of inputSource object, signal state and current input source.

244      *       <tr class="odd"><th>inputSourceList[].inputPort</th><th>String</th><th>input label (ext://hdmi:1, ext://hdmi:2, ext://dp:1, ext://dvi:1 etc) </th><th>required</th></tr>

247      *   </tbody>
248      * </table>
249      * </div>

251      * @example
252      * // Javascript code
253      * function getInputSourceStatus () {
254      *     
255      *     function successCb(cbObject) {
256      *         var inputSourceList = cbObject.inputSourceList;
257      *         for ( var i = 0; i < inputSourceList.length; i++) {
258      *             console.log("inputSourceList[" + i + "] : " + JSON.stringify(inputSourceList[i]));
259      *             console.log("inputSourceList[" + i + "].inputPort : " + inputSourceList[i].inputPort);
260      *         }
261      *         console.log("currentInputSource : " + cbObject.currentInputSource);
262      *         console.log("currentSignalState : " + cbObject.currentSignalState);
263      *         // Do something
264      *     }
265      *
266      *     function failureCb(cbObject) {
267      *         var errorCode = cbObject.errorCode;
268      *         var errorText = cbObject.errorText;
269      *         
270      *         console.log ("Error Code [" + errorCode + "]: " + errorText);
271      *     }
272      *
273      *     var inputSource = new InputSource();
274      *     inputSource.getInputSourceStatus(successCb, failureCb);
275      * }
276      * @since 1.0
277      * @see

279     InputSource.prototype.getInputSourceStatus = function (successCallback, errorCallback) {
280     
281         log("getInputSourceStatus: ");
282     
283         service.Request("luna://com.webos.service.eim/", {
284             method : "getAllInputStatus",
285             onSuccess : function(result) {
286                 log("getInputSourceStatus: On Success");
287 
288                 if (result.returnValue === true) {
289                     if (successCallback && typeof successCallback === 'function') {
290                         var cbObj = {};
291                         var inputList = new Array(result.totalCount);
292                         var convertList = new Array(result.totalCount);
293                         
294                         
295                         for (var i=0; i<inputList.length; i++) {
296                             inputList[i] = {};
297                             inputList[i].inputPort = parseInputSource(result.devices[i].label);
298                             
299                             // fixed input for hdmi:3, hdmi:4. requested by youngjun.lee
300                             var tempInputSource = result.devices[i].id.split("_");
301                             inputList[i].inputPort = 'ext://' + tempInputSource[0].toLowerCase() + ':' + tempInputSource[1];
302                             if (inputList[i].inputPort === 'ext://hdmi:3') {
303                                 inputList[i].inputPort = 'ext://dp:1';
304                             } else if (inputList[i].inputPort === 'ext://hdmi:4') {
305                                 inputList[i].inputPort = 'ext://dvi:1';
306                             }
307                             // end
308                             
309                             convertList[i] = {};
310                             convertList[i].inputPort = inputList[i].inputPort;
311                             convertList[i].id = result.devices[i].id;
312                         }
313                         
314                         cbObj.inputSourceList = inputList;
315                         
316                         service.Request("luna://com.webos.service.eim/", {
317                             method : "getCurrentInput",
318                             parameters : {},
319                             onSuccess : function(result) {
320                                 log("InputSource.getInputSourceStatus: On Success 3");
321                                 
322                                 if (result.returnValue === true) {
323                                     
324                                     if (successCallback && typeof successCallback === 'function') {
325                                         
326                                         cbObj.currentInputSource = {};
327                                         
328                                         for (var i=0; i<convertList.length; i++) {
329                                             if (convertList[i].id === result.mainInputSourceId) {
330                                                 cbObj.currentInputSource = convertList[i].inputPort;
331                                                 break;
332                                             }
333                                         }
334                                         
335                                         cbObj.currentSignalState = "unknown";
336                                         
337                                         if (broadcast !== null) {
338                                             log("InputSource.getInputSourceStatus : broadcast is not null");
339                                             
340                                             broadcast.getSignalStatus(function successCb(cbObject) {
341                                                 
342                                                 cbObj.currentSignalState = cbObject.videoSignalState;
343                                                 log("InputSource.getInputSourceStatus: On Success 2");
344 
345                                                 if (successCallback && typeof successCallback === 'function') {
346                                                     log("getInputSourceStatus: On Success" + JSON.stringify(cbObj));
347                                                     successCallback(cbObj);
348                                                     return;
349                                                 }
350                                                 
351                                             }, function failureCb() {
352                                                 
353                                                 log("InputSource.getInputSourceStatus : signal state is fail.");
354                                                 if (successCallback && typeof successCallback === 'function') {
355                                                     log("getInputSourceStatus: On Success" + JSON.stringify(cbObj));
356                                                     successCallback(cbObj);
357                                                     return;
358                                                 }
359                                             });
360                                             
361                                         } else {
362                                             
363                                             log("InputSource.getInputSourceStatus : it does not initialize.");
364                                             if (successCallback && typeof successCallback === 'function') {
365                                                 log("getInputSourceStatus: On Success" + JSON.stringify(cbObj));
366                                                 successCallback(cbObj);
367                                                 return;
368                                             }
369                                         }
370                                     }
371                                 }
372                             },
373                             onFailure : function(result) {
374                                 log("InputSource.getInputSourceStatus: On Failure 2");
375                                 delete result.returnValue;
376                                 if (errorCallback && typeof errorCallback === 'function') {
377                                     checkErrorCodeNText(result, "IGISS", "InputSource.getInputSourceStatus returns failure.");
378                                     errorCallback(result);
379                                     return;
380                                 }
381                             }
382                         });
383                     }
384                 }
385             },
386             onFailure : function(result) {
387                 log("getInputSourceStatus: On Failure");
388                 delete result.returnValue;
389                 if (errorCallback && typeof errorCallback === 'function') {
390                     checkErrorCodeNText(result, "IGISS", "InputSource.changeInputSource returns failure on gathering input list.");
391                     errorCallback(result);
392                 }
393             }
394         });
395                 
396     log("InputSource.getInputSourceStatus Done");

400     function parseInputSource(str) {
401         
402         var index = 1;
403         var length = str.length;
404         
405         if (!isNaN(parseInt(str.charAt(length-1), 10))) {
406             // has number
407             index = str.charAt(length-1);
408             length--;
409         }
410         
411         str = str.substring(0, length) +":"+ index;
412         str = "ext://" + str.toLowerCase();
413         
414         return str;
415     }

418     function convertInputSource(inputSource, successCallback, errorCallback) {
419         
420         service.Request("luna://com.webos.service.eim/", {
421             method : "getAllInputStatus",
422             onSuccess : function(result) {
423                 log("convertInputSource: On Success " + inputSource);
424 
425                 if (result.returnValue === true) {
426                     if (successCallback && typeof successCallback === 'function') {
427                         var inputList = new Array(result.totalCount);
428                         
429                         log("convertInputSource: " + JSON.stringify(result.totalCount));
430                         log("convertInputSource: " + JSON.stringify(result.devices));
431                         
432                         // fixed input for hdmi:3, hdmi:4. requested by youngjun.lee
433                         if (inputSource === 'ext://dp:1') {
434                             inputSource = 'ext://hdmi:3';
435                         } else if (inputSource === 'ext://dvi:1') {
436                             inputSource = 'ext://hdmi:4';
437                         }
438                         // end 
439                         
440                         for (var i=0; i<inputList.length; i++) {
441                             inputList[i] = {};
442                             inputList[i].inputPort = parseInputSource(result.devices[i].label);
443                             var tempInputSource = result.devices[i].id.split("_");
444                             inputList[i].id = 'ext://' + tempInputSource[0].toLowerCase() + ':' + tempInputSource[1];
445                             
446                             if (inputSource === inputList[i].id) {
447                                 // ok
448                                 log("convertInputSource: On Success ok " + inputSource);
449                                 successCallback(inputSource);
450                                 return;
451                             }
452                         }
453                         
454                         for (var j=0; j<inputList.length; j++) {
455                             if (inputSource === inputList[j].inputPort) {
456                                 // convert source
457                                 inputSource = inputList[j].id;
458                                 log("convertInputSource: On Success converted " + inputSource);
459                                 successCallback(inputSource);
460                                 return;
461                             }
462                         }
463                     }
464                 }
465             },
466             onFailure : function(result) {
467                 log("convertInputSource: On Failure");
468                 delete result.returnValue;
469                 if (errorCallback && typeof errorCallback === 'function') {
470                     checkErrorCodeNText(result, "IGISS", "convertInputSource returns failure on gathering input list.");
471                     errorCallback(result);
472                 }
473             }
474         });
