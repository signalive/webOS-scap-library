639 	function getSystemSettings(category, keys, cbObjectGetter, successCallback , errorCallback){
640 		var params = {
641 				category : category,
642 				keys : keys
643 		};
644 
645 		service.Request("luna://com.webos.service.commercial.signage.storageservice/settings/", {
646 			method : "get",
647 			parameters : params,
648 			onSuccess : function(result) {
649 				log("On Success");
650 				if(result.returnValue === true){
651 					var cbObj = cbObjectGetter(result.settings);
652 					if(cbObj === false){
653 						if(errorCallback && typeof errorCallback === 'function'){
654 							errorCallback({
655 								errorText: "Invalid DB value",
656 								errorCode: "DB_ERROR"
657 							});
658 						}
659 					}
660 					else{
661 						if(successCallback && typeof successCallback === 'function'){
662 							log("successCallback!!!!!!!!!");
663 							successCallback(cbObj);
664 						}
665 						else{
666 							log("successCallback not registered or is not a function: " + successCallback);
667 						}
668 					}
669 				}
670 				else{
671 					log("Settings Failed:  " + JSON.stringify(result, null, 3));
672 
673 					if(errorCallback && typeof errorCallback === 'function'){
674 						errorCallback({
675 							errorText: "Invalid DB value : " + result.errorText,
676 							errorCode: "DB_ERROR"
677 						});
678 					}
679 
680 				}
681 			},
682 			onFailure : function(result) {
683 				log("On Failure");
684 				delete result.returnValue;
685 				// See if any salvagable data
686 				if(result.settings){
687 					log("settings = " + JSON.stringify(result.settings, null, 3));
688 
689 					var cbObject = cbObjectGetter(result.settings);
690 
691 					log("errorKey = " + JSON.stringify(result.errorKey, null, 3));
692 
693 					for(var i=0;i < result.errorKey.length; ++i){
694 						cbObject[result.errorKey[i]] = Signage.UNDEFINED;
695 					}
696 
697 					log("cbObj = " + JSON.stringify(cbObject, null, 3));
698 					if(successCallback && typeof successCallback === 'function'){
699 						log("successCallback!!!!!!!!!");
700 						successCallback(cbObject);
701 					}
702 				}
703 				else {
704 					if(errorCallback && typeof errorCallback === 'function'){
705 						errorCallback({
706 							errorText:  ((typeof result.errorText === 'undefined') ? "DB Failure": result.errorText),
707 							errorCode: "DB_ERROR"
708 						});
709 					}
710 				}
711 			}
712 		});
713 
714 		log("Requested Service: " + "luna://com.webos.service.commercial.signage.storageservice/settings/");
715 		log("params : " + JSON.stringify(params));
716 	}
