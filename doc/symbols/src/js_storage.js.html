 32 	function parseUri (str) {
 33 		var	o   = parseUri.options,
 34 		m   = o.parser[o.strictMode ? "strict" : "loose"].exec(str),
 35 		uri = {},
 36 		i   = 14;
 37 
 38 		while (i--) {uri[o.key[i]] = m[i] || "";}
 39 
 40 		uri[o.q.name] = {};
 41 		uri[o.key[12]].replace(o.q.parser, function ($0, $1, $2) {
 42 			if ($1) {uri[o.q.name][$1] = $2;}
 43 		});
 44 		validateURI(uri);
 45 		return uri;
 46 	}
 47 
 48 	parseUri.options = {
 49 			strictMode: false,
 50 			key: ["source","protocol","authority","userInfo","user","password","host","port","relative","path","directory","file","query","anchor"],
 51 			q:   {
 52 				name:   "queryKey",
 53 				parser: /(?:^|&)([^&=]*)=?([^&]*)/g
 54 			},
 55 			parser: {
 56 				strict: /^(?:([^:\/?#]+):)?(?:\/\/((?:(([^:@]*)(?::([^:@]*))?)?@)?([^:\/?#]*)(?::(\d*))?))?((((?:[^?#\/]*\/)*)([^?#]*))(?:\?([^#]*))?(?:#(.*))?)/,
 57 				loose:  /^(?:(?![^:@]+:[^:@\/]*@)([^:\/?#.]+):)?(?:\/\/)?((?:(([^:@]*)(?::([^:@]*))?)?@)?([^:\/?#]*)(?::(\d*))?)(((\/(?:[^?#](?![^?#\/]*\.[^?#\/.]+(?:[?#]|$)))*\/?)?([^?#\/]*))(?:\?([^#]*))?(?:#(.*))?)/
 58 			}
 59 	};

 61 	function validateURI(parser){
 62 
 63 		// Take file, http, or https protocol only.
 64 		if(parser.protocol === 'file'){
 65 			if(parser.host === 'internal'){
 66 				return validateInternalURI(parser);
 67 			}
 68 			else if(parser.host === 'usb'){
 69 				validateUsbURI(parser);
 70 			}
 71 			else{
 72 				throw " Invalid Host: " + parser.host;
 73 			}
 74 		}
 75 		else if(parser.protocol === 'http' || parser.protocol === 'https'){
 76 			if(parser.path.charAt(parser.path.length-1) === '/'){
 77 				throw "Invalid path: " + parser.path;
 78 			}
 79 			else{
 80 				return;
 81 			}
 82 		}
 83 		else{
 84 			throw "Invalid protocol: " + parser.protocol;
 85 		}
 86 	}
 87 
 88 	function validateInternalURI(parser){
 89 
 90 		if(parser.path){
 91 			log("Path: " + parser.path);
 92 			if(parser.path.lastIndexOf("/") > 0){
 93 				throw "Invalid internal path: " + parser.path;
 94 			}
 95 			else{
 96 				return;
 97 			}
 98 		}
 99 		else{
100 			log("No Path: ");
101 			throw "Invalid internal path: " + parser.path;
102 		}
103 	}
104 
105 	function validateUsbURI(parser){
106 		if(parser.port){
107 			if(parser.port.match("/[0-9]+/")){
108 				return {result:true};
109 			}
110 			else{
111 				throw "Invalid USB host: " + parser.host;
112 			}
113 		}
114 		else{
115 			throw "Invalid USB host: " + parser.host;
116 		}
117 
118 		if(parser.path){
119 			if(parser.path.charAt(parser.path.length-1) === '/'){
120 				throw "Invalid USB path: " + parser.path;
121 			}
122 			else{
123 				return {result:true};
124 			}
125 		}
126 		else{
127 			throw "Invalid USB path: " + parser.path;
128 		}
129 	}

137     /**
138      * <p>
139      * Upgrades an application from the remote server to the local storage. The
140      * remote server information(serverIP and serverPort) are settled in a
141      * display menu. This new application will be launched at the next boot time
142      * and an old application will be deleted at the same time. An application
143      * on server should be located as like
144      * "http://serverIP:serverPort/procentric/scap/application/scap_app.zip".
145      * 
146      * Note : The size of scap_app.zip should be depending on storage size of
147      * platform. When the application upgrades, there are old application,
148      * scap_app.zip, content data stored by storage.copyFile API and new
149      * application extracted from scap_app.zip at the same time. So please
150      * consider about your application size to upgrade and check the amount of
151      * free space on your signage monitor by storage.getStorageInfo API.
152      * 
153      * </p>
154      * 
155      * @example
156      * 
157      * var successCb = function (cbObject){ console.log("Application Update
158      * successful"); };
159      * 
160      * var failureCb = function(cbObject){ var errorCode = cbObject.errorCode;
161      * var errorText = cbObject.errorText; console.log( " Error Code [" +
162      * errorCode + "]: " + errorText); };
163      * 
164      * storage = new Storage();
165      * storage.upgradeApplication(successCb, failureCb);
166      * 
167      * 
168      * @class storage
169      * @param {Function}
170      *            successCallback success callback function.
171      * @param {Function}
172      *            errorCallback failure callback function.
173      * @returns
174      *            <p>
175      *            After the method is successfully executed, successCallback is
176      *            called without any parameter.
177      *            </p>
178      *            <p>
179      *            If an error occurs while execution of this method,
180      *            errorCallback is called with errorCode and errorText.
181      *            </p>
182      * 
183      */
184 	Storage.prototype.upgradeApplication = function (successCallback, errorCallback) {
185 		
186 		service.Request("luna://com.webos.service.commercial.signage.storageservice", {
187 			method : "upgradeApplication",
188 			parameters : {
189 				from : 'remote'
190 			},
191 			onSuccess : function(result) {
192 				if(result.returnValue === true){
193 					successCallback();
194 				}
195 				else{
196 					errorCallback({errorCode:result.errorCode, errorText:result.errorText});
197 				}
198 			},
199 			onFailure : function(result) {
200 				errorCallback({errorCode:result.errorCode, errorText:result.errorText});
201 			}
202 		});
203

206 	/**
207 	 * <p>
208 	 * Copy a local file or remote file to the internal storage.
209 	 * File location is described in a URI form, where local file is always pointing to the 
210 	 * root directory of the internal memory.
211 	 * 
212 	 * No directory structure is supported for the internal storage, so no directory structure is allowed for the internal source URI.
213 	 * <p>
214 	 * Note : Only local application has access to local file which is copied. 
215 	 * 
216 	 * <p>
217 	 * <p>
218 	 * These URI schemes are available.
219 	 * <div align=left>
220 	 * <table class="hcap_spec" width=400>
221 	 *   <thead><tr><th>Name</th><th>Description</th></tr></thead>
222 	 *   <tbody>
223 	 *       <tr>
224 	 *       <th>file://internal/</th>
225 	 *       <th>Locate to a file in the internal memory space. 
226 	 *       Internal memory does not support directory structure, and no directory should be included in the path 
227 	 *       beyond the root path. All the files will be stored in the root path of the internal memory.
228 	 *       </th>
229 	 *       </tr>
230 	 *       <tr class="odd">
231 	 *       <th>file://usb:[index]/</th>
232 	 *       <th>Locate to a file in a USB storage device at USB port [index].</th>
233 	 *       </tr>
234 	 *       <tr>
235 	 *       <th>http://</th>
236 	 *       <th>Locate to a file in the remote location. Download using HTTP GET protocol.</th>
237 	 *       </tr>
238 	 *   </tbody>
239 	 * </table>
240 	 * </div>
241 	 * </p>
242 	 * <p>
243 	 * Copying a file from/to any paths other than the internal memory or coping a local file to a remote location
244 	 * are not allowed. A file which is downloaded can be accessed through "./content" folder of local application.
245 	 * </p>
246 	 * 
247 	 * @example 
248 	 * 
249 	 * var successCb = function (){
250 	 *	console.log("Copying File done.");
251 	 * }; 
252 	 * 
253 	 * var failureCb = function(cbObject){ 
254 	 *	var errorCode = cbObject.errorCode;
255 	 *	var errorText = cbObject.errorText; 
256 	 *	console.log( " Error Code [" + errorCode + "]: " + errorText); 
257 	 * };
258 	 * 
259 	 * // Copy a local file in the internal memory to another location in internal memory.
260 	 * var options_local_to_local = { 
261 	 *	source: 'file://internal/copy_me.txt',
262 	 *	destination : 'file://internal/here.txt'
263 	 * };
264 	 * 
265 	 * var storage = new Storage();
266 	 * storage.copyFile(successCb, failureCb, options_local_to_local);
267 	 * 
268 	 * // Copy a remote file to the internal memory.
269 	 * var options_remote_to_local = { 
270 	 *	source: 'http://remote.file.site/copy_me.txt',
271 	 *	destination : 'file://internal/here.txt'
272 	 * }; 
273 	 * 
274 	 * var storage = new Storage();
275 	 * storage.copyFile(successCb, failureCb, options_remote_to_local);
276 	 *          
277 	 * // Copy a file from a USB memory stick at port 1 to the internal memory.
278 	 * var options_usb_to_local = { 
279 	 *	source: 'file://usb:1/from/copy_me.txt',
280 	 *	destination : 'file://internal/here.txt'
281 	 * };
282 	 * 
283 	 * var storage = new Storage();
284 	 * storage.copyFile(successCb, failureCb, options_usb_to_local);
285 	 * 
286 	 * @class storage
287 	 * @param {Function}
288 	 *            successCallback success callback function.
289 	 * @param {Function}
290 	 *            errorCallback failure callback function.
291 	 * @param {Object}
292 	 *            options Input Parameters
293 	 * @param {String}
294 	 *            options.source URI to the source file. 
295 	 * @param {String}
296 	 *            options.destination. URI to the destination file. 
297 	 * @returns
298 	 *            <p>
299 	 *            After the method is successfully executed, successCallback is called
300 	 *            without any parameter.
301 	 *            </p>
302 	 *            <p>
303 	 *            If an error occurs while execution of this method, errorCallback
304 	 *            is called with errorCode and errorText.
305 	 *            </p>
306 	 * 
307 	 */
308 	Storage.prototype.copyFile = function (successCallback , errorCallback, options ) {
309 
310 		log("Options: " + JSON.stringify(options, null , 3));
311 
312 		if(typeof options ==='undefined' || typeof options.destination !== 'string' || typeof options.source !== 'string'){
313 			log("Bad options" );
314 			errorCallback({errorCode:"BAD_PARAM", errorText:JSON.stringify(options, null, 3)});
315 			return;
316 		}
317 
318 		service.Request("luna://com.webos.service.commercial.signage.storageservice", {
319 			method : "fs/copyFile",
320 			parameters : {dest : options.destination , src:options.source},
321 			onSuccess : function(result) {
322 				if(result.returnValue === true){
323 					log("SUCCESS");
324 					successCallback();
325 				}
326 				else{
327 					log("Err: " + result.errorText);
328 					errorCallback({errorCode:result.errorCode, errorText:result.errorText});
329 				}
330 			},
331 			onFailure : function(result) {
332 				log("Err: " + result.errorText);
333 				errorCallback({errorCode:result.errorCode, errorText:result.errorText});
334 			}
335 		});
336 	};
