 33 	function parseUri (str) {
 34 		var	o   = parseUri.options,
 35 		m   = o.parser[o.strictMode ? "strict" : "loose"].exec(str),
 36 		uri = {},
 37 		i   = 14;
 38 
 39 		while (i--) {uri[o.key[i]] = m[i] || "";}
 40 
 41 		uri[o.q.name] = {};
 42 		uri[o.key[12]].replace(o.q.parser, function ($0, $1, $2) {
 43 			if ($1) {uri[o.q.name][$1] = $2;}
 44 		});
 45 		validateURI(uri);
 46 		return uri;
 47 	}

 51 			key: ["source","protocol","authority","userInfo","user","password","host","port","relative","path","directory","file","query","anchor"],
 52 			q:   {
 53 				name:   "queryKey",
 54 				parser: /(?:^|&)([^&=]*)=?([^&]*)/g
 55 			},
 56 			parser: {

 58 				loose:  /^(?:(?![^:@]+:[^:@\/]*@)([^:\/?#.]+):)?(?:\/\/)?((?:(([^:@]*)(?::([^:@]*))?)?@)?([^:\/?#]*)(?::(\d*))?)(((\/(?:[^?#](?![^?#\/]*\.[^?#\/.]+(?:[?#]|$)))*\/?)?([^?#\/]*))(?:\?([^#]*))?(?:#(.*))?)/

 62 	function validateURI(parser){
 63 
 64 		// Take file, http, or https protocol only.
 65 		if(parser.protocol === 'file'){
 66 			if(parser.host === 'internal'){
 67 				return validateInternalURI(parser);
 68 			}
 69 			else if(parser.host === 'usb'){
 70 				validateUsbURI(parser);
 71 			}
 72 			else{
 73 				throw " Invalid Host: " + parser.host;
 74 			}
 75 		}
 76 		else if(parser.protocol === 'http' || parser.protocol === 'https'){
 77 			if(parser.path.charAt(parser.path.length-1) === '/'){
 78 				throw "Invalid path: " + parser.path;
 79 			}
 80 			else{
 81 				return;
 82 			}
 83 		}
 84 		else{
 85 			throw "Invalid protocol: " + parser.protocol;
 86 		}
 87 	}

 89 	function validateInternalURI(parser){
 90 
 91 		if(parser.path){
 92 			log("Path: " + parser.path);
 93 			if(parser.path.lastIndexOf("/") > 0){
 94 				throw "Invalid internal path: " + parser.path;
 95 			}
 96 			else{
 97 				return;
 98 			}
 99 		}
100 		else{
101 			log("No Path: ");
102 			throw "Invalid internal path: " + parser.path;
103 		}
104 	}

107 		if(parser.port){
108 			if(parser.port.match("/[0-9]+/")){
109 				return {result:true};

111 			else{
112 				throw "Invalid USB host: " + parser.host;
113 			}
114 		}
115 		else{
116 			throw "Invalid USB host: " + parser.host;
117 		}
118 
119 		if(parser.path){

121 				throw "Invalid USB path: " + parser.path;
122 			}
123 			else{
124 				return {result:true};
125 			}
126 		}
127 		else{
128 			throw "Invalid USB path: " + parser.path;
129 		}
