 42     function checkErrorCodeNText(result, errorCode, errorText) {
 43         
 44         if (result.errorCode === undefined || result.errorCode === null ) {
 45             result.errorCode = errorCode;
 46         }
 47         if (result.errorText ===undefined || result.errorText === null) {
 48             result.errorText = errorText;

 52     /**
 53      * Get video size. 
 54      * @class Video
 55      * @param {Function} successCallback success callback function.
 56      * @param {Function} errorCallback failure callback function.
 57      * @return {Object} 
 58      * <div align=left>
 59      * <table class="hcap_spec" width=400>
 60      *   <thead><tr><th>Property</th><th>Type</th><th>Description</th><th>Required</th></tr></thead>
 61      *   <tbody>
 62      *       <tr><th>source</th><th>Object</th><th>source rectangle of video. </th><th>required</th></tr>
 63      *       <tr><th>source.x</th><th>Number</th><th>The x coordinate of the upper-left corner of the rectangle. </th><th>required</th></tr>
 64      *       <tr class="odd"><th>source.y</th><th>Number</th><th>The y coordinate of the upper-left corner of the rectangle. </th><th>required</th></tr>
 65      *       <tr><th>source.width</th><th>Number</th><th>The width of the rectangle. </th><th>required</th></tr>
 66      *       <tr class="odd"><th>source.height</th><th>Number</th><th>The height of the rectangle. </th><th>required</th></tr>
 67      *   </tbody>
 68      * </table>
 69      *
 70      * @example
 71      * // Javascript code
 72      * function getVideoStatus () {
 73      *     function successCb(cbObject) {
 74      *         console.log("cbObject : " + JSON.stringify(cbObject.source));
 75      *
 76      *         console.log("source.x : " + cbObject.source.x);
 77      *         console.log("source.y : " + cbObject.source.y);
 78      *         console.log("source.width : " + cbObject.source.width);
 79      *         console.log("source.height : " + cbObject.source.height);
 80      *         // Do something
 81      *         ...
 82      *     }
 83      *
 84      *     function failureCb(cbObject) {
 85      *         var errorCode = cbObject.errorCode;
 86      *         var errorText = cbObject.errorText;
 87      *         
 88      *         console.log ("Error Code [" + errorCode + "]: " + errorText);
 89      *     }
 90      *
 91      *     var video = new Video();
 92      *     video.getVideoStatus(successCb, failureCb);
 93      * }
 94      * @since 1.0
 95      * @see
 96      * <a href="Video%23setVideoSize.html">Video.setVideoSize()</a><br>
 97      */
 98     Video.prototype.getVideoStatus = function (successCallback, errorCallback) {
 99 
100         log("getVideoStatus: ");
101         
102         service.Request("luna://com.webos.service.tv.signage/", {
103             method : "getVideoSize",
104             onSuccess : function(result) {
105                 log("getVideoStatus: On Success");
106                 
107                 if (result.returnValue === true) {
108                     if (successCallback && typeof successCallback === 'function') {
109                         var cbObj = {};
110                         cbObj.source = result.videoSize.source;
111                         successCallback(cbObj);
112                     }
113                 }
114             },
115             onFailure : function(result) {
116                 log("getVideoStatus: On Failure");
117                 delete result.returnValue;
118                 if (errorCallback && typeof errorCallback === 'function') {
119                     checkErrorCodeNText(result, "VGVS", "Video.getVideoStatus returns failure.");
120                     errorCallback(result);
121                 }
122             }
123         });
124         
125         log("Video.getVideoStatus Done");
126     
127     };

131     /**
132      * Set video size.<br>
133      * Original video(source rectangle) will be adjusted to the full screen size. So it can resize or apply a zoom in effect to the video.
134      * 
135      * Note : We recommend to use just one video tag in html page. Even though an html page has one more video tag, one video source will be played.
136      *   
137      * @class Video
138      * @param {Function} successCallback success callback function.
139      * @param {Function} errorCallback failure callback function.
140      * @param {Object} options source object vary depending on video resolution.  
141      *                  And rectangle object has boundary value as : <br>
142      *                  source.x + source.width <= width of video resolution , source.y + source.height <= height of video resolution <br>
143      *                  x and y value of rectangle are upper than zero. 
144      * <br><image src="./image/setVideoSize.png"><br>
145      * <div align=left>
146      * <table class="hcap_spec" width=400>
147      *   <thead><tr><th>Property</th><th>Type</th><th>Description</th><th>Required</th></tr></thead>
148      *   <tbody>
149      *       <tr><th>source</th><th>Object</th><th>source rectangle of video. </th><th>required</th></tr>
150      *       <tr><th>source.x</th><th>Number</th><th>The x coordinate of the upper-left corner of the rectangle. </th><th>required</th></tr>
151      *       <tr class="odd"><th>source.y</th><th>Number</th><th>The y coordinate of the upper-left corner of the rectangle. </th><th>required</th></tr>
152      *       <tr><th>source.width</th><th>Number</th><th>The width of the rectangle. </th><th>required</th></tr>
153      *       <tr class="odd"><th>source.height</th><th>Number</th><th>The height of the rectangle. </th><th>required</th></tr>
154      *   </tbody>
155      * </table>
156      * 
157      * @return If the method is successfully executed, call the success callback function without a parameter. If an error occurs while performing the method, the failure callback function is called and Failure Callback Object is passed as a parameter.
158      * @example
159      * // Javascript code
160      * function setVideoSize () {
161      *     var options = {source: {x:10, y:10, width:900, height:600}};
162      *     
163      *     function successCb() {
164      *         // Do something
165      *     }
166      *
167      *     function failureCb(cbObject) {
168      *         var errorCode = cbObject.errorCode;
169      *         var errorText = cbObject.errorText;
170      *         
171      *         console.log ("Error Code [" + errorCode + "]: " + errorText);
172      *     }
173      *
174      *     var video = new Video();
175      *     video.setVideoSize(successCb, failureCb, options);
176      * }
177      * @since 1.0
178      * @see
179      * <a href="Video%23getVideoStatus.html">Video.getVideoStatus()</a><br>
180      * <a href="InputSource%23initialize.html#constructor">InputSource.initialize()</a><br>
181      */
182     Video.prototype.setVideoSize = function (successCallback, errorCallback, options) {
183     
184         log("setVideoSize: " + JSON.stringify(options));
185         
186         if (options.source === undefined ||
187                 typeof options.source.x !== 'number' || typeof options.source.y !== 'number' ||
188                 typeof options.source.width !== 'number' || typeof options.source.height !== 'number' ||
189                 options.source.x < 0 || options.source.y < 0 || options.source.width <= 0 || options.source.height <= 0 ) {
190             
191             if (errorCallback && typeof errorCallback === 'function') {
192                 var result = {};
193                 checkErrorCodeNText(result, "VSVS", "Video.setVideoSize returns failure. out of range or type error.");
194                 errorCallback(result);
195             }
196             
197             return;
198         }
199                 
200         service.Request("luna://com.webos.service.tv.signage/", {
201             method : "getVideoSize",
202             onSuccess : function(result) {
203                 log("setVideoSize: On Success");
204                 
205                 if (result.returnValue === true) {
206                     
207                     var cbObj = {};
208                     cbObj.x = result.videoSize.destination.x;
209                     cbObj.y = result.videoSize.destination.y;
210                     cbObj.width = result.videoSize.destination.width;
211                     cbObj.height = result.videoSize.destination.height;
212                     
213                     //console.log("current video size : " + JSON.stringify(result.videoSize.source));
214                     //console.log("saved video : " + JSON.stringify(Video.currentVideo));
215                 
216                     var videoTags = document.getElementsByTagName("video"); // jshint ignore:line
217                     var findVideoTags = false;
218                     for (var i=0; i<videoTags.length; i++) {
219                         //console.log("video tag: " + videoTags[i]);
220                         //console.log("video tag:id " + videoTags[i].id);
221                         // video tag
222                         if (videoTags[i].currentTime > 0) {
223                             //console.log("video tag is working");
224                             findVideoTags = true;
225                             if (Video.currentVideo.uri !== videoTags[i].src || 
226                                     (videoTags[i].id !== null && videoTags[i].id !== undefined && Video.currentVideo.tagId !== null && Video.currentVideo.tagId !== undefined && Video.currentVideo.tagId !== videoTags[i].id) ) {
227                                 Video.currentVideo.uri = videoTags[i].src;
228                                 Video.currentVideo.source = result.videoSize.source;
229                                 Video.currentVideo.tagId = videoTags[i].id;
230                                 //console.log("saved new video : " + JSON.stringify(Video.currentVideo));
231                             }
232                             break;
233                         }
234                     }
235                     if(findVideoTags === false) {
236                         // external input
237                         service.Request("luna://com.webos.service.eim/", {
238                             method : "getCurrentInput",
239                             parameters : {},
240                             onSuccess : function(resultw) {
241                                 //console.log("current input " + JSON.stringify(resultw));
242                                 
243                                 if (resultw.returnValue === true && Video.currentVideo.uri !== resultw.mainInputSourceId ||
244                                         (videoTags[0].id !== null && videoTags[0].id !== undefined && Video.currentVideo.tagId !== null && Video.currentVideo.tagId !== undefined && Video.currentVideo.tagId !== videoTags[0].id) ) {
245                                     Video.currentVideo.uri = resultw.mainInputSourceId;
246                                     Video.currentVideo.tagId = (videoTags[0] !== null && videoTags[0].id !== null && videoTags[0].id !== undefined ? videoTags[0].id : null);
247                                     service.Request("luna://com.webos.service.tv.signage/", {
248                                         method : "getVideoSize",
249                                         onSuccess : function(resultx) {
250                                             log("setVideoSize: On Success 1");
251                                             
252                                             if (resultx.returnValue === true) {
253                                                 
254                                                 Video.currentVideo.source = resultx.videoSize.source;
255                                                 //console.log("saved new video : " + JSON.stringify(Video.currentVideo));
256                                                 
257                                                 if (resultx.videoSize.source.width === 0 && resultx.videoSize.source.height === 0) {
258                                                     
259                                                     Video.currentVideo = {uri : null, source : null, tagId : null};
260                                                     
261                                                     var resultCallBack = {};
262                                                     checkErrorCodeNText(resultCallBack, "VSVS", "Video.setVideoSize returns failure. Not ready to setVideoSize." );
263                                                     errorCallback(resultCallBack);
264                                                     return;
265                                                 } else if ( Video.currentVideo.uri === null || Video.currentVideo.source === null || 
266                                                         (options.source.width + options.source.x) > (Video.currentVideo.source.x + Video.currentVideo.source.width) ||
267                                                         (options.source.height + options.source.y) > (Video.currentVideo.source.y + Video.currentVideo.source.height) ) {
268                                                     
269                                                     var resultCallBack2 = {};
270                                                     checkErrorCodeNText(resultCallBack2, "VSVS", "Video.setVideoSize returns failure. out of range or type error. (" + Video.currentVideo.source.width + " : " + Video.currentVideo.source.height + ")");
271                                                     errorCallback(resultCallBack2);
272                                                     return;
273                                                     
274                                                 }
275                                                 
276                                                 service.Request("luna://com.webos.service.tv.signage/", {
277                                                     method : "setVideoSize",
278                                                     parameters : {
279                                                         videoSize : {  "source": {
280                                                                             "x": options.source.x,
281                                                                             "y": options.source.y,
282                                                                             "width": options.source.width,
283                                                                             "height": options.source.height
284                                                                         },
285                                                                         "destination": {
286                                                                             "x": cbObj.x,
287                                                                             "y": cbObj.y,
288                                                                             "width": cbObj.width,
289                                                                             "height": cbObj.height
290                                                                         }
291                                                                     }
292                                                     },
293                                                     onSuccess : function(result) {
294                                                         log("setVideoSize: On Success 2");
295                                                         
296                                                         if (result.returnValue === true && successCallback && typeof successCallback === 'function') {
297                                                             successCallback();
298                                                             return;
299                                                         }
300                                                     },
301                                                     onFailure : function(result) {
302                                                         log("setVideoSize: On Failure 2");
303                                                         delete result.returnValue;
304                                                         if (errorCallback && typeof errorCallback === 'function') {
305                                                             checkErrorCodeNText(result, "VSVS", "Video.setVideoSize returns failure. Can't current video source size.");
306                                                             errorCallback(result);
307                                                             return;
308                                                         }
309                                                     }
310                                                 });
311                                                 
312                                             }
313                                         }
314                                     });
315                                 } else {
316                                     if ( Video.currentVideo.uri === null || Video.currentVideo.source === null ||
317                                             (options.source.width + options.source.x) > (Video.currentVideo.source.x + Video.currentVideo.source.width)||
318                                             (options.source.height + options.source.y) > (Video.currentVideo.source.y + Video.currentVideo.source.height) ){
319                                         
320                                         var resultCallBack = {};
321                                         checkErrorCodeNText(resultCallBack, "VSVS", "Video.setVideoSize returns failure. out of range or type error. (" + Video.currentVideo.source.width + " : " + Video.currentVideo.source.height + ")");
322                                         errorCallback(resultCallBack);
323                                         return;
324                                     }
325                                     
326                                     service.Request("luna://com.webos.service.tv.signage/", {
327                                         method : "setVideoSize",
328                                         parameters : {
329                                             videoSize : {  "source": {
330                                                                 "x": options.source.x,
331                                                                 "y": options.source.y,
332                                                                 "width": options.source.width,
333                                                                 "height": options.source.height
334                                                             },
335                                                             "destination": {
336                                                                 "x": cbObj.x,
337                                                                 "y": cbObj.y,
338                                                                 "width": cbObj.width,
339                                                                 "height": cbObj.height
340                                                             }
341                                                         }
342                                         },
343                                         onSuccess : function(result) {
344                                             log("setVideoSize: On Success 3");
345                                             
346                                             if (result.returnValue === true && successCallback && typeof successCallback === 'function') {
347                                                 successCallback();
348                                                 return;
349                                             }
350                                         },
351                                         onFailure : function(result) {
352                                             log("setVideoSize: On Failure 3");
353                                             delete result.returnValue;
354                                             if (errorCallback && typeof errorCallback === 'function') {
355                                                 checkErrorCodeNText(result, "VSVS", "Video.setVideoSize returns failure. Can't current video source size.");
356                                                 errorCallback(result);
357                                                 return;
358                                             }
359                                         }
360                                     });
361                                 } 
362                             },
363                             onFailure : function() {}
364                         });
365                     } else {
366                         
367                         if ( Video.currentVideo.uri === null || Video.currentVideo.source === null ||
368                                 (options.source.width + options.source.x) > (Video.currentVideo.source.x + Video.currentVideo.source.width) ||
369                                 (options.source.height + options.source.y) > (Video.currentVideo.source.y + Video.currentVideo.source.height) ) {
370                             
371                             var resultCallBack = {};
372                             checkErrorCodeNText(resultCallBack, "VSVS", "Video.setVideoSize returns failure. out of range or type error. (" + Video.currentVideo.source.width + " : " + Video.currentVideo.source.height + ")");
373                             errorCallback(resultCallBack);
374                             return;
375                         }
376                         
377                         service.Request("luna://com.webos.service.tv.signage/", {
378                             method : "setVideoSize",
379                             parameters : {
380                                 videoSize : {  "source": {
381                                                     "x": options.source.x,
382                                                     "y": options.source.y,
383                                                     "width": options.source.width,
384                                                     "height": options.source.height
385                                                 },
386                                                 "destination": {
387                                                     "x": cbObj.x,
388                                                     "y": cbObj.y,
389                                                     "width": cbObj.width,
390                                                     "height": cbObj.height
391                                                 }
392                                             }
393                             },
394                             onSuccess : function(result) {
395                                 log("setVideoSize: On Success 4");
396                                 
397                                 if (result.returnValue === true && successCallback && typeof successCallback === 'function') {
398                                     successCallback();
399                                     return;
400                                 }
401                             },
402                             onFailure : function(result) {
403                                 log("setVideoSize: On Failure 4");
404                                 delete result.returnValue;
405                                 if (errorCallback && typeof errorCallback === 'function') {
406                                     checkErrorCodeNText(result, "VSVS", "Video.setVideoSize returns failure. Can't current video source size.");
407                                     errorCallback(result);
408                                     return;
409                                 }
410                             }
411                         });
412                     }
413                     
414                     
415                 }
416             },
417             onFailure : function(result) {
418                 log("setVideoSize: On Failure");
419                 delete result.returnValue;
420                 if (errorCallback && typeof errorCallback === 'function') {
421                     checkErrorCodeNText(result, "VSVS", "Video.setVideoSize returns failure.");
422                     errorCallback(result);
423                     return;
424                 }
425             }
426         });
427         
428         log("Video.setVideoSize Done");
429     };
