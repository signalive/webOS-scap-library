 40     /**
 41      * Get network information. Returns the information including IP address assigned to wireless / wired LAN, gateway address, netmask address and DNS address.
 42      * @class DeviceInfo
 43      * @param {Function} successCallback success callback function.
 44      * @param {Function} errorCallback failure callback function.
 45      * @return {Object} 
 46      * <div align=left>
 47      * <table class="hcap_spec" width=400>
 48      *   <thead><tr><th>Property</th><th>Type</th><th>Description</th><th>Required</th></tr></thead>
 49      *   <tbody>
 50      *       <tr><th>isInternetConnectionAvailable</th><th>Boolean</th><th>Internet connection true: connected, false: not connected</th><th>required</th></tr>
 51      *       <tr class="odd"><th>wired</th><th>Object</th><th>Wired network information object. </th><th>required</th></tr>
 52      *       <tr><th>wired.state</th><th>String</th><th>Wired network connection "connected": connected "disconnected": not connected</th><th>required</th></tr>     
 53      *       <tr class="odd"><th>wired.interfaceName</th><th>String</th><th>Wired network interface name. </th><th>required</th></tr>
 54      *       <tr><th>wired.ipAddress</th><th>String</th><th>IP address assigned to the wired network interface. </th><th>required</th></tr>
 55      *       <tr class="odd"><th>wired.netmask</th><th>String</th><th>Netmask address assigned to the wired network interface. </th><th>required</th></tr>
 56      *       <tr><th>wired.gateway</th><th>String</th><th>Gateway address assigned to the wired network interface</th><th>required</th></tr>
 57      *       <tr class="odd"><th>wired.onInternet</th><th>String</th><th>Internet connection to the wired network "yes": connected "no": not connected. </th><th>required</th></tr>     
 58      *       <tr><th>wired.method</th><th>String</th><th>IP allocation "fixed": fixed allocation "dhcp": dynamic allocation</th><th>required</th></tr>
 59      *       <tr class="odd"><th>wired.dns1</th><th>String</th><th>DNS address assigned to the wired network interface. </th><th>required</th></tr>     
 60      *       <tr><th>wired.dns2</th><th>String</th><th>Secondary DNS address assigned to the wired network interface. </th><th>required</th></tr>     
 61      *       <tr class="odd"><th>wifi</th><th>Object</th><th>Wireless network information object. </th><th>required</th></tr>     
 62      *       <tr><th>wifi.state</th><th>String</th><th>Wireless network connection "connected": connected "disconnected": not connected</th><th>required</th></tr>     
 63      *       <tr class="odd"><th>wifi.interfaceName</th><th>String</th><th>Wireless network interface name. </th><th>required</th></tr>
 64      *       <tr><th>wifi.ipAddress</th><th>String</th><th>IP address assigned to the wireless network interface. </th><th>required</th></tr>
 65      *       <tr class="odd"><th>wifi.netmask</th><th>String</th><th>Netmask address assigned to the wireless network interface. </th><th>required</th></tr>
 66      *       <tr><th>wifi.gateway</th><th>String</th><th>Gateway address assigned to the wireless network interface</th><th>required</th></tr>
 67      *       <tr class="odd"><th>wifi.onInternet</th><th>String</th><th>Internet connection to the wireless network "yes": connected "no": not connected. </th><th>required</th></tr>
 68      *       <tr><th>wifi.method</th><th>String</th><th>IP allocation "fixed": fixed allocation "dhcp": dynamic allocation</th><th>required</th></tr>
 69      *       <tr class="odd"><th>wifi.dns1</th><th>String</th><th>DNS address assigned to the wireless network interface. </th><th>required</th></tr>
 70      *       <tr><th>wifi.dns2</th><th>String</th><th>Secondary DNS address assigned to the wireless network interface. </th><th>required</th></tr>
 71      *   </tbody>
 72      * </table>
 73      * </div>
 74      *
 75      * @example
 76      * // Javascript code
 77      * function getSoundStatus () {
 78      *     function successCb(cbObject) {
 79      *         console.log("cbObject : " + JSON.stringify(cbObject));
 80      *         console.log("isInternetConnectionAvailable : " + cbObject.isInternetConnectionAvailable);
 81      *         console.log("wired.state : " + cbObject.wired.state);
 82      *         console.log("wired.method : " + cbObject.wired.method);
 83      *         console.log("wired.ipAddress : " + cbObject.wired.ipAddress);
 84      *         console.log("wired.netmask : " + cbObject.wired.netmask);
 85      *         console.log("wired.dns1 : " + cbObject.wired.dns1);
 86      *         console.log("wired.dns2 : " + cbObject.wired.dns2);
 87      *         console.log("wifi.state : " + cbObject.wifi.state);
 88      *         console.log("wifi.method : " + cbObject.wifi.method);
 89      *         console.log("wifi.ipAddress : " + cbObject.wifi.ipAddress);
 90      *         console.log("wifi.netmask : " + cbObject.wifi.netmask);
 91      *         console.log("wifi.dns1 : " + cbObject.wifi.dns1);
 92      *         console.log("wifi.dns2 : " + cbObject.wifi.dns2);
 93      *     }
 94      *
 95      *     function failureCb(cbObject) {
 96      *         var errorCode = cbObject.errorCode;
 97      *         var errorText = cbObject.errorText;
 98      *         
 99      *         console.log ("Error Code [" + errorCode + "]: " + errorText);
100      *     }
101      *
102      *     var deviceInfo = new DeviceInfo();
103      *     deviceInfo.getNetworkInfo(successCb, failureCb);
104      * }
105      * @since 1.0
106      * @see
107      * <a href="DevieInfo%23getNetworkMacInfo.html">DeviceInfo.getNetworkMacInfo()</a><br>
108      */
109     DeviceInfo.prototype.getNetworkInfo = function (successCallback, errorCallback) {
110     
111         log("getNetworkInfo: ");
112     
113         service.Request('luna://com.palm.connectionmanager', {
114             method: 'getstatus',
115             parameters: {},
116             onSuccess: function(result) {
117                 log("getNetworkInfo: onSuccess");
118                 delete result.returnValue;
119                 successCallback && successCallback(result);
120             },
121             onFailure: function(error) {
122                 log("getNetworkInfo: onFailure");
123                 delete error.returnValue;
124                 errorCallback && errorCallback(error);
125             }
126         });
127         
128         log("DeviceInfo.getNetworkInfo Done");
129     };

197     /**
198      * Get the platform information of device. Platform information includes manufacturer, model name, software version, hardware version and product serial number.
199      * @class DeviceInfo
200      * @param {Function} successCallback success callback function.
201      * @param {Function} errorCallback failure callback function.
202      * @return {Object} 
203      * <div align=left>
204      * <table class="hcap_spec" width=400>
205      *   <thead><tr><th>Property</th><th>Type</th><th>Description</th><th>Required</th></tr></thead>
206      *   <tbody>
207      *       <tr><th>hardwareVersion</th><th>String</th><th>Hardware version of webOS TV. </th><th>required</th></tr>     
208      *       <tr class="odd"><th>manufacturer</th><th>String</th><th>Manufacturer of webOS TV. </th><th>required</th></tr>
209      *       <tr><th>modelName</th><th>String</th><th>webOS TV model name. </th><th>required</th></tr>     
210      *       <tr class="odd"><th>sdkVersion</th><th>String</th><th>SDK version of webOS TV. </th><th>required</th></tr>
211      *       <tr><th>serialNumber</th><th>String</th><th>webOS TV serial number. </th><th>required</th></tr>
212      *       <tr class="odd"><th>firmwareVersion</th><th>String</th><th>Firmware version of webOS TV. </th><th>required</th></tr>
213      *   </tbody>
214      * </table>
215      * </div>
216      *
217      * @example
218      * // Javascript code
219      * function getPlatformInfo () {
220      *     function successCb(cbObject) {
221      *         console.log("cbObject : " + JSON.stringify(cbObject));
222      *         console.log("hardwareVersion : " + cbObject.hardwareVersion);
223      *         console.log("modelName : " + cbObject.modelName);
224      *         console.log("sdkVersion : " + cbObject.sdkVersion);
225      *         console.log("serialNumber : " + cbObject.serialNumber);
226      *         console.log("firmwareVersion : " + cbObject.firmwareVersion);
227      *
228      *         // Do something
229      *         ...
230      *     }
231      *
232      *     function failureCb(cbObject) {
233      *         var errorCode = cbObject.errorCode;
234      *         var errorText = cbObject.errorText;
235      *         
236      *         console.log ("Error Code [" + errorCode + "]: " + errorText);
237      *     }
238      *
239      *     var deviceInfo = new DeviceInfo();
240      *     deviceInfo.getPlatformInfo(successCb, failureCb);
241      * }
242      * @since 1.0
243      * @see
244      * <a href="DevieInfo%23getNetworkMacInfo.html">DeviceInfo.getNetworkMacInfo()</a><br>
245      */
246     DeviceInfo.prototype.getPlatformInfo = function (successCallback, errorCallback) {
247     
248         log("getPlatformInfo: ");
249         
250         service.Request('luna://com.webos.service.tv.systemproperty', {
251             method: 'getSystemInfo',
252             parameters: {
253                 keys: ["modelName", "serialNumber", "firmwareVersion", "hardwareVersion", "sdkVersion"]
254             },
255             onSuccess: function(result) {
256                 log("getPlatformInfo: onSuccess");
257                 result.manufacturer = "LGE";
258                 result.sdkVersion = "1.0.21000";
259                 delete result.returnValue;
260                 delete result.errorText;
261                 successCallback && successCallback(result);
262             },
263             onFailure: function(error) {
264                 log("getPlatformInfo: onFailure");
265                 delete error.returnValue;
266                 errorCallback && errorCallback(error);
267             }
268         });
269         
270         log("DeviceInfo.getPlatformInfo Done");
271     };
