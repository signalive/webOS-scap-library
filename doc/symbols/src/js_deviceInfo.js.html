270     /**
271      * Gets device usage information. Usage information includes cpu and memory status.
272      * @class DeviceInfo
273      * @param {Function} successCallback success callback function.
274      * @param {Function} errorCallback failure callback function.
275      * @param {Object} options
276      * <div align=left>
277      * <table class="hcap_spec" width=400>
278      *   <thead><tr><th>Property</th><th>Type</th><th>Description</th><th>Required</th></tr></thead>
279      *   <tbody>
280      *       <tr><th>cpus</th><th>Boolean</th><th>true to get CPU information, false otherwise.</th><th>optional</th></tr>
281      *       <tr class="odd"><th>memory</th><th>Boolean</th><th>true to get memory information, false otherwise.</th><th>optional</th></tr>
282      *   </tbody>
283      * </table>
284      * </div>
285      * @return {Object} 
286      * <div align=left>
287      * <table class="hcap_spec" width=400>
288      *   <thead><tr><th>Property</th><th>Type</th><th>Description</th><th>Available</th></tr></thead>
289      *   <tbody>
290      *       <tr><th>cpus</th><th>Array</th><th>Array of objects containing information about each CPU/core installed: model, and times <br>(an object containing the number of milliseconds the CPU/core spent in: user, nice, sys, idle, and irq).</th><th>optional</th></tr>
291      *       <tr class="odd"><tr><th>memory</th><th>Object</th><th>Object containing total and free member fields. <br>'total' is total amount of system memory in bytes and 'free' is amount of free memory in bytes.</th><th>optional</th></tr>     
292      *   </tbody>
293      * </table>
294      * </div>
295      *
296      * @example
297      * // Javascript code
298      * function getSystemUsageInfo() {
299      *   function successCb(cbObject) {
300      *      console.log("cbObject : " + JSON.stringify(cbObject));
301      *      console.log("memory.total : " + cbObject.memory.total);
302      *      console.log("memory.free : " + cbObject.memory.free);
303      *
304      *      for (var i in cbObject.cpus) {
305      *         console.log("cpu.model " +  cbObject.cpus[i].model);
306      *         console.log("cpu.times.user " +  cbObject.cpus[i].times.user);
307      *         console.log("cpu.times.nice " +  cbObject.cpus[i].times.nice);
308      *         console.log("cpu.times.sys " +  cbObject.cpus[i].times.sys);
309      *         console.log("cpu.times.idle " +  cbObject.cpus[i].times.idle);
310      *         console.log("cpu.times.irq " +  cbObject.cpus[i].times.irq);
311      *      } 
312      *      // Do something
313      *   }
314      *
315      *   function failureCb(cbObject) {
316      *      var errorCode = cbObject.errorCode;
317      *      var errorText = cbObject.errorText;
318      *      console.log ("Error Code [" + errorCode + "]: " + errorText);
319      *   }
320      *
321      *   var deviceInfo = new DeviceInfo();
322      *   var options = {cpus : true, memory : true};
323      *   deviceInfo.getSystemUsageInfo(successCb, failureCb, options);
324      * }
325      * @since 1.2
326      * @see
327      * <a href="DevieInfo%23getPlatformInfo.html">DeviceInfo.getPlatformInfo()</a><br>
328      */
329     DeviceInfo.prototype.getSystemUsageInfo = function (successCallback, errorCallback, options) {
330         
331         log("getSystemUsageInfo: ");
332         
333         service.Request("luna://com.webos.service.commercial.signage.storageservice", {
334             method : "getSystemUsageInfo",
335             parameters : {
336                 cpus : options.cpus,
337                 memory : options.memory
338             },
339             onSuccess : function(result) {
340                 if (result.returnValue === true) {
341                     var ret = {};
342                     if (typeof result.memory !== 'undefined' ) {
343                     	ret.memory = result.memory;
344                     }
345                     if (typeof result.cpus !== 'undefined' ) {
346                         ret.cpus = result.cpus;
347                     }
348                     successCallback(ret);
349                 } else {
350                     errorCallback({errorCode:result.errorCode, errorText:result.errorText});
351                 }
352             },
353             onFailure : function (result) {
354                 errorCallback({errorCode:result.errorCode, errorText:result.errorText});
355             }
356         });
357         
358         log("DeviceInfo.getSystemUsageInfo Done");
359     };
